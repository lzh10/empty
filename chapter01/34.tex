\begin{verse}
    \section{十二月三十一日}
    \vspace*{2\ccwd}
    \Large\kaishu{
        灯火退尽

        年夕

        一个我

        躲暖窝

        风寒屏隔

        以外朗朗空明

        团栾留照

        点点星芒

        冬之夜

        夜之意趣尽然焉

        薄被覆身温软

        困意遐游

        迟迟不与枕合

        想

        昨昔历历

        算别处欢宵

        悠悠去此

        市声阗阗

        灯彩焕烂

        者般辉煌如旧

        今夕何夕

        有此百念

        百念纭纷

        如此良夜何

        ...

        众相去散

        复我梦来

        沉安
  
    }
\end{verse}
