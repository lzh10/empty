\begin{verse}
    \section{北天}
    \vspace*{2\ccwd}
    \Large\kaishu{
        日轮耀晔

        云一无

        天空清明简净

        形如扣碗

        青蓝色平匀

        缓缓淡下

        浅白已至碗沿

        四围密合

        莽苍苍大地

        冬木肃肃挺立

        细枝映裂灰纹

        叶消尽

        道旁林阴疏落

        午时走一程

        风泠然砭骨

        诚不如晨行之冽

        往折十余分钟

        食宿奔波横越省际

        暖服寒透

        入室复暖

        如此数度

        闲日轻轻即过
  
    }
\end{verse}
