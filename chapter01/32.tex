\begin{verse}
    \section{慢步}
    \vspace*{2\ccwd}
    \Large\kaishu{
        南来的薰风

        空气中丝丝甜香

        青草刚刈过

        微涩味感踅入鼻腔

        \vspace*{2\ccwd}

        小园里的错落景致

        昨冬移植的树只有干

        个个扭展怪异手势

        人所不能拗出的

        \vspace*{2\ccwd}

        大多是绿，深绿、浅绿、杂灰的绿

        还有别的色，黄的一丛紫的一簇

        余者孑然不成群了

        散零在草隙树荫墙角

        \vspace*{2\ccwd}

        阳光扑在身上

        如被思念者拥进怀中

        一个人款款款款走

        真愿就这样永恒下去
  
    }
\end{verse}
