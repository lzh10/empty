\begin{verse}
    \section{五月}
    \vspace*{2\ccwd}
    \Large\kaishu{
        五月的阵雨、阴霾与微风

        远处的马达、警哨与汽鸣

        在此一刻相遇

        我见如慢放的镜头

        归巢的燕、摇曳的树、行人的伞

        于时空中定格

        以暖湿的空气为布

        映成一长卷朦胧的春色

        在下一刻轮复

        久长 久长
        
        \vspace*{2\ccwd}

        尔后

        让这细雨如丝可落檐成珠

        让这繁枝茂叶都华滋新发
        
        让这尘嚣万千再无与相扰
    }
\end{verse}
