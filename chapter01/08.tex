\begin{verse}
    \section{风雪无归}
    \vspace*{2\ccwd}
    \Large\kaishu{
        街灯莹莹

        远影蒙蒙

        大雪风飘

        漫天纷扬

        那糯白的、沁凉的、破碎的

        万瓣千片

        都迎向我、融化我、遮没我

        天因此寒呐

        \vspace*{2\ccwd}

        灌丛簇连

        披戴鹅绒

        行道茫茫

        人何茕茕

        看旋坠的、冰结的、覆叠的

        绵绵不绝

        且裹掖绿、掩蔽土、禁缚声

        夜如此长呵
        
        \vspace*{2\ccwd}

        归路其远哟
        
        倩谁人捎载还家
    }
\end{verse}
