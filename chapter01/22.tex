\begin{verse}
    \section{破我}
    \vspace*{2\ccwd}
    \Large\kaishu{
        是重幕相阻隔

        是荆丛为掩碍

        狭剑刺不透的

        奋引阔斧劈擘

        野火灼不尽的

        聚萃炽焰焚灭

        我以心力开新路

        入得眼来尽旧雠

        杀敌且壮行

        九十九度割喉的酒还未够烈

        请斟一碗炙鼎七日不涸滚滚沸嚣的血

        沛然豪莽的臆气破胸荡漫卷万丈的尘

        莫要祝那凯还的祷词

        大风萧萧——

        曲须激昂唱和一襟赴死的慨慷

        
        \hspace*{2\ccwd}敌茫茫兮心决然

        
        \hspace*{2\ccwd}戮不罄兮身不返

        
        \hspace*{2\ccwd}红潋潋兮血盈海

        
        \hspace*{2\ccwd}别青天兮踏黄泉
    }
\end{verse}
