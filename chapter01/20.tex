\begin{verse}
    \section{如梦恋}
    \vspace*{2\ccwd}
    \Large\kaishu{
        想梦中最是执著

        思不通变转

        唯求不可预之尽头

        若牵心需涉水

        不思量其千丈也游渡

        若意怀欲翻山

        不疑犹便万仞也越绝

        一根筋也似的愚

        直待死亡将我拉离

        只现实何能——

        何能令我爱你义无反顾

        至死才休 \@ 至死而不休

        我畏难甚于惧死

        畏失望甚于绝望

        我畏太无谓

        我爱亦然焉

        于幽昧之陬角暗窥所恋

        自以为之恋 \@ 匪恋

        滥情应笑我 \@ 卑渺如虫豸
    }
\end{verse}
