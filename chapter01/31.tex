\begin{verse}
    \section{熬夜}
    \vspace*{2\ccwd}
    \Large\kaishu{
        薄雾轻寒

        夜阑声未断

        蚊营营

        灯煌煌

        嗡嗡扇转

        邀同此间试无眠

        漫漫

        凭窗纳取光万点

        \vspace*{2\ccwd}

        拼却明朝

        且吟今宵

        共声色纷挠

        心浮飞飘

        骀荡随风青天上

        遥遥

        捱到五更后

        待日晞

        睡云晓
    }
\end{verse}
