\begin{verse}
    \section{朝日}
    \vspace*{2\ccwd}
    \Large\kaishu{
       曦起

       东方之若明

       林雾晓色迷濛

       溪泉声清灵或隐

       有闻鸟语啭鸣

       小兽怀眠欲醒

       叶隙间开微光落进

       花身慵倦似美人持颐

       青翠参差托悬粒粒珠莹

       蕈盖丛丛

       昨已朽故的木精

       风餐露饮

       腐土里茁新

       饱蓄爱意的演绎

       朝日临 \@ 万类生息
    }
\end{verse}
