\begin{verse}
    \section{复醒}
    \vspace*{2\ccwd}
    \Large\kaishu{
        醒来

        已复第二度

        已逾零时

        已去一日

        已而卒岁

        （呼——）

        神思回明

        撷来一片

        记得正欢浓

        哪！夜色太浅太浅

        遮不住~喔

        漫兴而肆的情流

        且共昨我语、频频

        无动于中的

        百转千回

        越淡漠的越萦肠

        最疏冷的最热望

        哦！如可偿愿

        许淋漓荒梦一场

        嚣噪匝地的宣吐

        红尘接天的吹扬

        一念起唐皇

        一霎那

        泯没同如天光

        此寂寥

        此无星月的晴空

        光点尽已撒落

        街衢升起

        更辉耀的芒

        短瞬的盛放

        永恒之思啊 

        昏黄隐灭

        平明昭然东方

    }
\end{verse}
