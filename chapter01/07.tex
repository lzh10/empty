\begin{verse}
    \section{雪夜}
    \vspace*{2\ccwd}
    \Large\kaishu{
        天什么时候晴

        月什么时候明

        风雪什么时候悄然落停

        屋檐什么时候挂结冰凌

        这内里的切望就要满盈

        这外间精灵妆扮着仙境

        雪夜梦幻呐如灼思萦
        
        \vspace*{2\ccwd}

        在等云消在等雪霁

        在等月半在等月圆

        在等昼日喧嚣归于沉寂

        在等霜花次第蔓过窗扉

        在等灯火如晕熏染的雪色温莹里

        一场不期的邂逅和无声的错别

        留下我——

        渐长的身影 \@ 消入茫茫夜色无寻

    }
\end{verse}
