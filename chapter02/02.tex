\begin{verse}
    \section{我来此地}
    \vspace*{2\ccwd}
    \Large\kaishu{
        我来此地

        未再有过安眠

        回忆的幽梦

        没入 夏夜的喧嚣

        \vspace*{2\ccwd}

        心自离乡

        再未有处安放

        远寻的孤舟

        遗落 风雨中飘摇

        \vspace*{2\ccwd}

        情至相思

        未曾有在预料

        绵延的丝念

        游离 十指间缠绕

        \vspace*{2\ccwd}

        天意所枉

        未能有从愿怀

        尘世的无常

        堪笑 命宿里挣逃
    }
\end{verse}
