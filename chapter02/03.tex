\begin{verse}
    \section{秋怀}
    \vspace*{2\ccwd}
    \Large\kaishu{
        时深雨凉

        不见满目萧黄\footnote{南方的秋，此刻仍是草木葱茂}

        故里野荒

        望极草长风欢\footnote{想像应如是}

        何言回寻

        相思无归\footnote{故乡已不是家}

        对冷秋轻语

        一一成滴露\footnote{言说不尽的思念凝成湿露}

        \vspace*{2\ccwd}
        
        长道行远

        唯今还顾

        旧知曾识终不复\footnote{旧物旧人都已抛空}

        我应有所恋

        系萦心怀

        吹尽花落或返\footnote{但余念端，落红归落有处}

        犹想炊烟

        是最无着的依眷\footnote{可见而无可触，是以心心而无行}
    }
\end{verse}
