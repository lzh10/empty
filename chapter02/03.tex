\begin{verse}
    \section{秋怀}
    \vspace*{2\ccwd}
    \Large\kaishu{
        时深雨凉

        不见满目萧黄

        故里野荒

        望极草长风欢

        何言回寻

        相思无归

        对冷秋轻语

        一一成滴露

        \vspace*{2\ccwd}
        
        长道行远

        唯今还顾

        旧知曾识终不复

        我应有所恋

        系萦心怀

        吹尽花落或返

        犹想炊烟

        是最无着的依眷
    }
\end{verse}
