\begin{verse}
    \section{遐}
    \vspace*{2\ccwd}
    \Large\kaishu{
        丛翠摇欢，

        残红萧乱，
        
        紫霞襟连向晚。

        甚风意辗转，
        
        如何昨个都不怜，

        今来惜看，

        一一缺痕数点。

        犹也似、别后牵怨，

        中有几千山。
        
        恨相思不到，

        书凭秋雁，

        目断只到天云边。

        天云边，

        芳音杳绝，

        秋声亘远。
    }
\end{verse}
