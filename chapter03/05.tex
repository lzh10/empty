\begin{verse}
    \section{喜欢}
    \vspace*{2\ccwd}
    \Large\kaishu{
        遇着一个人

        蓦然地喜欢

        喜欢她的漂亮、她的活泼

        还有逸发光采的灵性

        她清晨的笑是初阳

        向晚的笑是云霞

        晴夜的笑是月华

        就算是雨天呵

        细雨绵绵又离离

        都似她不经勾撩耳后的丝发

        那么些美好 \@ 让我觉自卑

        因我像个孤老头子

        未经红尘 \@ 已然沧桑

        颓在枯木旁倚候

        一个恰如春天的人

        遇见 \@ 蓦然就喜欢
    }
\end{verse}
